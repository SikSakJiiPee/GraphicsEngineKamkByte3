#version 130
//Vertex shader operoi jokaisella verteksill채

//input data VBO:lta. Jokainen verteksi on 2 floatia
in vec2 vertexPosition;
in vec4 vertexColor;
in vec2 vertexUV;

out vec2 fragmentPosition;
out vec4 fragmentColor;
out vec2 fragmentUV;

uniform mat4 P;

void main()
{
	//Asetetaan x,y positio ruudulle
	gl_Position.xy = vertexPosition; //tsekkaa t채m채 viel채
	//z positio on nolla, koska 2D
	gl_Position.z = 0.0;
	
	//Osoittaa koordinaattien olevan normalisoitu
	gl_Position.w = 1.0;
	
	fragmentPosition = vertexPosition;
	
	fragmentColor = vertexColor;
	
	fragmentUV = vec2 (vertexUV.x, 1.0 - vertexUV.y);
}
